# Пример настройки crontab для автоматической инкрементальной синхронизации
# 
# Для установки выполните: crontab -e
# Затем добавьте одну из строк ниже:

# Каждые 30 минут
*/30 * * * * /Users/evgenijsikunov/projects/beton-crm/beton-crm/server/src/scripts/runIncrementalSyncCron.sh

# Каждые 2 часа
0 */2 * * * /Users/evgenijsikunov/projects/beton-crm/beton-crm/server/src/scripts/runIncrementalSyncCron.sh

# Каждые 6 часов
0 */6 * * * /Users/evgenijsikunov/projects/beton-crm/beton-crm/server/src/scripts/runIncrementalSyncCron.sh

# Ежедневно в 2:00 ночи
0 2 * * * /Users/evgenijsikunov/projects/beton-crm/beton-crm/server/src/scripts/runIncrementalSyncCron.sh

# Ежедневно в 2:00 и 14:00
0 2,14 * * * /Users/evgenijsikunov/projects/beton-crm/beton-crm/server/src/scripts/runIncrementalSyncCron.sh

# Только в рабочие дни (понедельник-пятница) в 9:00
0 9 * * 1-5 /Users/evgenijsikunov/projects/beton-crm/beton-crm/server/src/scripts/runIncrementalSyncCron.sh

# Каждые 15 минут в рабочее время (9:00-18:00, понедельник-пятница)
*/15 9-18 * * 1-5 /Users/evgenijsikunov/projects/beton-crm/beton-crm/server/src/scripts/runIncrementalSyncCron.sh

# 
# Примечания:
# - Замените путь на актуальный путь к вашему проекту
# - Убедитесь, что скрипт имеет права на выполнение: chmod +x runIncrementalSyncCron.sh
# - Логи будут сохраняться в logs/incremental-sync-cron.log
# - Для просмотра логов: tail -f logs/incremental-sync-cron.log
#
# Для проверки текущих задач crontab: crontab -l
# Для удаления всех задач crontab: crontab -r
